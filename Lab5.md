### Makefile:

~~~
 block.o:  
    cc block.c -c -o block.o -fPIC  
    cc -shared block.o -o block.so  
    ar qc block.a block.o  
    cc ../program.c block.a -o static_block.out  
    cc ../program.c block.so -o dynamic_block.out -Wl,-rpath .

 clean:  
    rm \*.o  
    rm \*.a  
    rm \*.so  
    rm \*.out  
~~~
